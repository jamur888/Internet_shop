create table admin (

id BIGINT PRIMARY KEY,
login varchar,
password varchar

);

create table client (

id BIGINT PRIMARY KEY,
name varchar,
login varchar,
password varchar,
address varchar,
phone bigint

);

Create table product(

id BIGINT PRIMARY KEY,
name varchar,
amount integer,
price FLOAT,

);


Create table "order"(

id BIGINT PRIMARY KEY,
number BIGINT,
client_id BIGINT references client(id),
product_id BIGINT references product(id)

);
